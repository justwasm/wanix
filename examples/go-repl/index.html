package main import ( "crypto/sha256" "fmt" "io" "os" ) func main() { f, err := os.Open("hello.wasm") if err != nil { panic(err) } defer f.Close() h := sha256.New() if _, err := io.Copy(h, f); err != nil { panic(err) } fmt.Printf("%x\n", h.Sum(nil)) }
